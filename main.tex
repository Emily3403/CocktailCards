\def\ml#1{%
    \pgfmathsetmacro\result{#1 * 0.0333}%0.03814}%
    \pgfmathprintnumberto[precision=2]{\result}{\result}%
    #1 ml (\result~oz.)%
}
\def\counter{1}

% #1 = name, #2 = ingredients (pgffor), #3 = Equipment (pgffor), #4 = Method (pgffor)
\def\cocktail#1#2#3#4{
    \vspace{15pt}
    %
    \begin{center}
    {\bf#1}
    \end{center}
    \textbf{Ingredients:}
    \begin{itemize}
        \foreach \x in {#2} {%
            \item \x
        }
    \end{itemize}
    %
    \vspace{5pt}
    \textbf{Equipment: }%
    \foreach[count=\i] \x in {#3}{\ifnum\i=1\else, \fi\x}%
    %
    \vspace*{7pt}%
    \\%
    \textbf{Method:}
    \begin{enumerate}
        \foreach \x in {#4} {%
            \item \x
        }
    \end{enumerate}
    %
    \vspace{20pt}
    \newpage
}
