    \begin{minipage}[t]{0.49\textwidth}%

        \begin{itemize}
            \foreach \x in {#2} {%
                \item \x
            }
        \end{itemize}
    \end{minipage}
    %
    \begin{minipage}[t]{0.49\textwidth}%
        \vspace{0pt}
        \begin{enumerate}
            \foreach \x in {#4} {%
                \item \x
            }
        \end{enumerate}
    \end{minipage}

    \ifempty{#3}
    \else
    \medskip
    \begin{center}
        \textsl{Tools: \foreach[count=\i] \x in {#3}{%
            \ifnum%
            \i=1\else%
            , %
            \fi\x}}%
    \end{center}
    \fi
}
